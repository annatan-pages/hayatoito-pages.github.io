.. code-block:: java

        int N = a.length;
        for (int i = 0; i < N; i++) {
            for (int j = N-2; j >= i; j--) {
                if (a[j+1].compareTo(a[i]) < 0) {
                    swap(a, j, j+1);
                    swap(b, j, j+1);
                }
            }
        }
    }
